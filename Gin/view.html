    获取请求路径：c.Request.URL

import (
	"fmt"
	"github.com/gin-gonic/gin"
)

func test(c *gin.Context) {
	// 打印一些基本的请求信息
	fmt.Println("请求方式：", c.Request.Method)
	fmt.Println("请求路径：", c.Request.URL)
	c.String(200, "test")
}

func main() {
	r := gin.Default()
	r.GET("/test", test)
	r.Run()

    获取请求IP

    package main

import (
	"fmt"
	"github.com/gin-gonic/gin"
)

func test(c *gin.Context) {
	// 打印一些基本的请求信息
	fmt.Println("请求方式：", c.Request.Method)
	fmt.Println("请求路径：", c.Request.URL)
	fmt.Println("请求头：", c.Request.Header)
	fmt.Println("请求头的信息：", c.GetHeader("User-Agent"))
	// 获取IP
	fmt.Println("请求IP：", c.ClientIP())        // IP
	fmt.Println("请求IP：", c.RemoteIP())        // IP
	fmt.Println("请求IP：", c.Request.RemoteAddr)  // IP+Port

	c.String(200, "test")
}

func main() {
	r := gin.Default()
	r.GET("/test", test)
	r.Run()
}

    # 获取post请求数据

    c.PostForm("键") 如果有值则返回值，没值则返回空

    c.DefaultPostForm("键", "默认值") 如果有键，则返回值，如果没键则返回默认值

    c.GetPostForm("键") 如果有值，则返回值和true，如果没值，则返回false

    package main

import (
	"fmt"
	"github.com/gin-gonic/gin"
)

func test(c *gin.Context) {
	// 请求数据的获取
	fmt.Println(c.PostForm("num"))
	fmt.Println(c.DefaultPostForm("num", "默认值"))
	fmt.Println(c.GetPostForm("num"))

	c.String(200, "test")
}

func main() {
	r := gin.Default()
	r.POST("/test", test)
	r.Run()
}

    # 路径参数

    :键名

    package main

import "github.com/gin-gonic/gin"

func test(c *gin.Context) {
	y := c.Param("year")
	m := c.Param("mouth")
	c.String(200, y+m)
}

func main() {
	r := gin.Default()
	r.GET("test/:year/:mouth", test)
	r.Run()
}

    # ShouldBind函数

    有什么作用？

    前面使用的PostForm方法没办法获取到json数据，而ShouldBind则可以通过结构体接收多种类型的数据。

    示例：

    package main

import (
	"fmt"
	"github.com/gin-gonic/gin"
	"log"
)

type Person struct {
	Name   string `json:"name"`
	Gender string `json:"gender"`
}

func test(c *gin.Context) {
	human := Person{}
	err := c.ShouldBind(&human)
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(human)
	c.String(200, "OK")
}

func main() {
	r := gin.Default()
	r.POST("/demo", test)
	r.Run()
}

    传入json数据

    这里需要注意一个点，定义结构体时，结构体中的字段名称需要大写，否则是不能使用的，因为在go语言中，大写才是外界可以读取到的。传值的json数据中并没有大写是因为在定义结构体时给字段在以json数据传输时重新命名了。
