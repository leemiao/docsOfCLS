    # 基本数据类型

    # 整型和浮点型

import (
    "fmt"
    "reflect"
)

func main() {
    var x = 10
    fmt.Println(x, reflect.TypeOf(x))
}

    var f float32
f = 3.14
fmt.Println(f, reflect.TypeOf(f))


    var g = 3.14
fmt.Println(g, reflect.TypeOf(g))

    # 索引

    索引从0开始

    # 字符串索引取值

    package main

import "fmt"

func main() {
    var s string
    s = "hello world"
    fmt.Println(s)
    
    // 基本操作
    // 索引
    fmt.Println(string(s[0]))
}

    # 字符串索引切片

    package main

import "fmt"

func main() {
    var s string
    s = "hello world"
    // fmt.Println(s)
    
    // 基本操作
    // 索引
    fmt.Println(string(s[0:2]))
}

    包含前面不包含后面

    fmt.Println(string(s[:2]))

    从头开始切片的话，可以省略前面的索引

    package main

import "fmt"

func main() {
    var s string
    s = "hello world"
    //fmt.Println(s)
    
    // 基本操作
    // 索引
    //fmt.Println(string(s[:2]))
    fmt.Println(string(s[2:]))
}

    切片到末尾的话，可以省略末尾的索引

    # 字符串内置方法

    package main

import (
    "fmt"
    "reflect"
    "strings"
)

func main() {
    var q = " Hello World "
    // 全部字母转大写
    var w = strings.ToUpper(q)
    fmt.Println(q)
    fmt.Println(w)
    // 全部转小写
    var e = strings.ToLower(q)
    fmt.Println(e)
    // Contains包含  字符串中是否包含xx字符串
    var r = strings.Contains(q, "ll")
    fmt.Println(r)
    // HasPrefix 字符串是否以...开头
    var t = strings.HasPrefix(q, "H")
    fmt.Println(t)
    // HasSuffix 字符串是否以...结尾
    var y = strings.HasSuffix(q, "l")
    fmt.Println(y)
    // Trim  去除字符串左右两端的空格
    var u = strings.Trim(q, " ")
    fmt.Println(u)
    // 输入
    //var i string
    //fmt.Scan(&i)
    //fmt.Println(i)
    
    // strings.Split 字符串拆分
    var actors = "雷电将军 枫源万叶 神里凌华"
    fmt.Println(len(actors)) // 38
    var actor = strings.Split(actors, " ")
    fmt.Println(actor)                 // [雷电将军 枫源万叶 神里凌华]
    fmt.Println(reflect.TypeOf(actor)) // []string 数组
    // strings.Join 字符串拼接
    var actorJ = strings.Join(actor, "-")
    fmt.Println(actorJ)
    // strings.Index 查询
    var actorI = strings.Index(actors, "雷")
    fmt.Println(actorI)
}

    # 类型转换

    go语言只支持强制类型转换

    package main

import (
    "fmt"
    "reflect"
    "strconv"
)

func main() {
    // 数据类型转换
    var a int8 = 8
    var b = int64(a)
    fmt.Println(b, reflect.TypeOf(b))
    var c = float32(a)
    fmt.Println(c, reflect.TypeOf(c))
    
    // 数字转字符串
    var d = 10
    var e = strconv.Itoa(d)
    fmt.Println(d, reflect.TypeOf(e))
    
    // 数字字符串转整型
    var j = "20"
    var k, _ = strconv.Atoi(j)
    fmt.Println(k, reflect.TypeOf(k))
    
    // 浮点型字符串转浮点型
    var f = "3.14"
    var l, _ = strconv.ParseFloat(f, 64)
    fmt.Println(l, reflect.TypeOf(l))
}
